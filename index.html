DOM Manipulatie - Complete Gids

Wat is de DOM?

De Document Object Model (DOM) is een representatie van je HTML pagina. JavaScript kan de DOM gebruiken om elementen te selecteren, te wijzigen, toe te voegen of te verwijderen.

1. Elementen Selecteren

// Selecteer één element
let element = document.querySelector('#mijnId');
let element2 = document.querySelector('.mijnClass');

// Selecteer meerdere elementen
let elementen = document.querySelectorAll('.mijnClass');
let alleDivs = document.querySelectorAll('div');

// Oude methodes (nog steeds bruikbaar)
let byId = document.getElementById('mijnId');
let byClass = document.getElementsByClassName('mijnClass');
let byTag = document.getElementsByTagName('div');

2. Content Wijzigen

// Tekst wijzigen
element.textContent = "Nieuwe tekst";
element.innerText = "Nieuwe tekst (met styling)";
element.innerHTML = "Bold tekst";

// Attributen wijzigen
element.setAttribute('id', 'nieuweId');
element.setAttribute('class', 'nieuweClass');
let waarde = element.getAttribute('id');

3. Styling Wijzigen

// Inline styles
element.style.color = "red";
element.style.backgroundColor = "blue";
element.style.fontSize = "20px";
element.style.display = "none"; // Verberg element
element.style.display = "block"; // Toon element

// CSS classes
element.classList.add('nieuweClass');
element.classList.remove('oudeClass');
element.classList.toggle('toggleClass');
element.classList.contains('mijnClass'); // true/false

4. Elementen Maken & Toevoegen

// Nieuw element maken
let nieuwElement = document.createElement('div');
nieuwElement.textContent = "Ik ben nieuw!";
nieuwElement.className = "mijnClass";

// Toevoegen aan DOM
document.body.appendChild(nieuwElement);
element.appendChild(nieuwElement);
element.insertBefore(nieuwElement, anderElement);

// Prependen (vooraan toevoegen)
element.insertAdjacentHTML('beforebegin', '
Voor element
');
element.insertAdjacentHTML('afterbegin', '
In element (begin)
');
element.insertAdjacentHTML('beforeend', '
In element (eind)
');
element.insertAdjacentHTML('afterend', '
Na element
');

5. Elementen Verwijderen

// Element verwijderen
element.remove(); // Moderne manier
element.parentNode.removeChild(element); // Oude manier

// Child verwijderen
element.removeChild(childElement);

// Alle children verwijderen
while (element.firstChild) {
    element.removeChild(element.firstChild);
}

// Of sneller:
element.innerHTML = '';

6. Traversing (Navigeren)

// Parent element
let parent = element.parentElement;
let parentNode = element.parentNode;

// Children
let children = element.children; // Alleen elementen
let childNodes = element.childNodes; // Inclusief tekst nodes

// Siblings (broers/zussen)
let nextSibling = element.nextElementSibling;
let previousSibling = element.previousElementSibling;

// Eerste/laatste child
let firstChild = element.firstElementChild;
let lastChild = element.lastElementChild;

7. Formulieren

// Form element selecteren
let form = document.querySelector('form');
let input = document.querySelector('input[type="text"]');

// Waarde krijgen/zetten
let waarde = input.value;
input.value = "Nieuwe waarde";

// Checkbox/Radio
let checkbox = document.querySelector('input[type="checkbox"]');
checkbox.checked = true;
let isChecked = checkbox.checked;

// Select dropdown
let select = document.querySelector('select');
select.value = "optie2";
let selectedValue = select.value;

8. Event Delegation

// Event listener op parent (efficiënter)
let container = document.querySelector('.container');
container.addEventListener('click', function(e) {
    if (e.target.classList.contains('button')) {
        console.log('Button geklikt!');
    }
});

// Event object gebruiken
element.addEventListener('click', function(event) {
    event.preventDefault(); // Voorkom default gedrag
    event.stopPropagation(); // Stop event bubbling
    console.log(event.target); // Element dat geklikt is
    console.log(event.type); // "click"
});
